Header <boost/plugin/alias.hpp>

Includes alias methods and macro. You can include this header or boost/plugin/shared_library.hpp to reduce dependencies in case you do not use the refcountable functions. 


BOOST_PLUGIN_ALIAS(FunctionOrVar, AliasName)
namespace boost {
  namespace plugin {
    template<typename T> 
      T & alias(const shared_library &, const boost::string_ref &);
  }
}

Header <boost/plugin/shared_library.hpp>

Contains the boost::plugin::shared_library class, main class for all the DLL/DSO operations. 

namespace boost {
  namespace plugin {
    class shared_library;
    bool operator==(const shared_library &, const shared_library &);
    bool operator!=(const shared_library &, const shared_library &);
    bool operator<(const shared_library &, const shared_library &);
    void swap(shared_library &, shared_library &);
  }
}

Header <boost/plugin/shared_library_load_mode.hpp>

namespace boost {
  namespace plugin {
    struct load_mode;
    load_mode::type operator|(load_mode::type, load_mode::type);
    load_mode::type & 
    operator|=(load_mode::type & left, load_mode::type right);
  }
}

Shared Library Refcountable Reference

Header <boost/plugin/refcountable.hpp>

Header <boost/plugin/refcountable.hpp>

Includes all the reference counting functions that hold a shared pointer to the instance of boost::plugin::shared_library. Those methods use boost::shared_ptr<T> and boost::function<T> classes. 

namespace boost {
  namespace plugin {
    template<typename T> 
      boost::function< T > 
      shared_function(const boost::shared_ptr< shared_library > &, 
                      boost::string_ref);
    template<typename T> 
      boost::function< T > 
      shared_function(const boost::filesystem::path &, boost::string_ref);
    template<typename T> 
      boost::function< T > 
      shared_function_alias(const boost::shared_ptr< shared_library > &, 
                            boost::string_ref);
    template<typename T> 
      boost::function< T > 
      shared_function_alias(const boost::filesystem::path &, 
                            boost::string_ref);
    template<typename T> 
      boost::shared_ptr< T > 
      shared_variable(const boost::shared_ptr< shared_library > &, 
                      boost::string_ref);
    template<typename T> 
      boost::shared_ptr< T > 
      shared_variable(const boost::filesystem::path &, boost::string_ref);
    template<typename T> 
      boost::shared_ptr< T > 
      shared_variable_alias(const boost::shared_ptr< shared_library > &, 
                            boost::string_ref);
    template<typename T> 
      boost::shared_ptr< T > 
      shared_variable_alias(const boost::filesystem::path &, 
                            boost::string_ref);
  }
}
